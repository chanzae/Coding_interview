```python
def solution(X, A):
    xlist = [i for i in range(1,X+1)]

    for idx in range(len(A)):
        if A[idx] in xlist:
            xlist.pop(xlist.index(A[idx]))

        if len(xlist) == 0:
            answer = idx
            break
        else:
            answer = -1
            
    return answer
```

```python
def solution(X, A):
    from collections import deque
    xlist = []
    adeque = deque(A)

    for i in range(len(A)):
        x = adeque.popleft()
        if x not in xlist:
            xlist.append(x)

        if len(xlist) == X:
            answer = i
            break
        else:
            answer = -1
            
    return answer
```
time complexity O(N ** 2) : 54%
